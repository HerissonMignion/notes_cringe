	transitif (fonction qui prend un ou plusieurs arguments)
	intransitif (fonction qui ne prend aucun argument)
	pronominal (verbe avec se, me, te, ...)
	impersonnel (verbe qui se conjugue avec le pronom impersonnel "il")
	attributif ((ex: verbe être) utilisé pour qualifier quelqu'un/chose)
	auxiliaire
		auxiliaire de conjugaison
		auxiliaire de modalité
		auxiliaire d'aspect
		auxiliaire factitif (notion avancée)
